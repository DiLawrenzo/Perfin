CREATE TABLE `budgetbook`.`expense` ( `id` INT(50) NOT NULL AUTO_INCREMENT , `amount` DOUBLE(50) NOT NULL , `category` VARCHAR(200) NOT NULL , `account` VARCHAR(200) NOT NULL , `ptitle` VARCHAR(200) NOT NULL , `payee` VARCHAR(50) NOT NULL , `periodic` ENUM('once','daily','regular','monthly') NOT NULL DEFAULT 'once' , `job_id` INT(5) NOT NULL , PRIMARY KEY (`id`(5))) ENGINE = InnoDB;
